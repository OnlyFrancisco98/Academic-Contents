$hostname = "localhost";
$username = "root";        // Usuario por defecto en xampp
$password = "";            // Contraseña vacía
$database = "mi_sitio_clashroyale";  // Mi BD

// Crear la conexión con MySQL
$mysqli = new mysqli($hostname, $username, $password, $database);

// Verificar si hubo error
if ($mysqli->connect_error) {
    echo "Error de conexión PHP-MySQL: " . $mysqli->connect_error;
} 
else {
    echo "Conexión exitosa a la base de datos <br>";

    // Ejemplo
    $sql = "SELECT * FROM products";
    echo "Ejecutando consulta: " . $sql . "<br>";

    $resultado = $mysqli->query($sql);

    if($resultado && $resultado->num_rows > 0){
        echo "Sí hay productos registrados<br>";

        // Obtener primer registro como ejemplo
        $fila = $resultado->fetch_array();
        echo "Primer producto: ID=".$fila['id'].", Nombre=".$fila['name']."<br>";
    }
    else {
        echo "No se encontraron productos<br>";
    }

    // Cerrar la conexión
    $mysqli->close();
}
